
```bash
sudo nano /etc/systemd/system/whisper.service
```

```bash
[Unit]
Description=Whisper FastAPI service
After=network.target

[Service]
User=user
Group=user
WorkingDirectory=/var/www/whisper_test
Environment="PATH=/var/www/whisper_test/.venv/bin"
ExecStart=/var/www/whisper_test/.venv/bin/uvicorn app:app --host 0.0.0.0 --port 8002 --workers 1
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target
```

```bash
sudo systemctl daemon-reload
sudo systemctl enable whisper
sudo systemctl start whisper
sudo systemctl status whisper
```
