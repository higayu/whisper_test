```bash
sudo nano /etc/systemd/system/media-converter.service
```

# 入力内容
```bash
  GNU nano 7.2               /etc/systemd/system/media-converter.service                         
# /etc/systemd/system/media-converter.service
[Unit]
Description=FFmpeg Media Converter API
After=network.target

[Service]
User=user
Group=user
WorkingDirectory=/var/www/whisper_test
Environment="PATH=/var/www/whisper_test/.venv/bin:/usr/bin"
ExecStart=/var/www/whisper_test/.venv/bin/python -m uvicorn convert_server:app --host 0.0.0.0 -->
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target
```

# 再起動
```bash
sudo systemctl daemon-reload
sudo systemctl enable media-converter
sudo systemctl start media-converter
sudo systemctl status media-converter
```
